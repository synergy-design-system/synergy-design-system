
## Default

Tabs are used inside tab groups to represent and activate tab panels.

```html
<syn-tab
  role="tab"
  aria-disabled="false"
  aria-selected="false"
  id="syn-tab-16"
  panel=""
  tabindex="0"
>
  Tab
</syn-tab>

<syn-tab
  active=""
  role="tab"
  aria-disabled="false"
  aria-selected="true"
  id="syn-tab-13"
  panel=""
  tabindex="0"
  >Active</syn-tab
>
<syn-tab
  closable=""
  role="tab"
  aria-disabled="false"
  aria-selected="false"
  id="syn-tab-14"
  panel=""
  tabindex="0"
  >Closable</syn-tab
>
<syn-tab
  disabled=""
  role="tab"
  aria-disabled="true"
  aria-selected="false"
  id="syn-tab-15"
  panel=""
  tabindex="-1"
  >Disabled</syn-tab
>

```
